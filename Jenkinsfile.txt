pipeline {
    agent any
    triggers {
        pollSCM '* * * * *'
            }

    stages {
        stage('Checkout') {
            steps {
                git branch: 'main', url: 'https://github.com/vijaykumarsdt/jgsu-spring-petclinic.git'
            }
        }
		
		stage('Build') {
            steps {
                bat './mvnw clean package'
            }
        }
		}
	
    post { 
        always { 
            junit '**/target/surefire-reports/TEST-*.xml'
                    archiveArtifacts 'target/*.jar'
        }
    }
}
